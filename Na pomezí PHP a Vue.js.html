    ApplicationEvents::ON_REQUEST => KernelEvents::REQUEST,

  • contributte/psr7-http-message PSR-7 (HTTP Message Interface) to Nette Framework

    public function onRequest(RequestEvent $event, $name, EventDispatcherInterface $dispatcher): void
{
    $psr7 = \Contributte\Psr7\Psr7ServerRequestFactory::fromNette(\Nette\Http\IRequest);
    $psr7 = $psr7->withApplicationRequest(\Nette\Application\Request);

    $factory = new \Symfony\Bridge\PsrHttpMessage\Factory\HttpFoundationFactory();
    $request = $factory->createRequest($psr7);

    $symfonyEvent = new GetResponseEvent($this->kernel, $request, HttpKernel::MASTER_REQUEST);
    $dispatcher->dispatch(KernelEvents::REQUEST, $symfonyEvent);
}
